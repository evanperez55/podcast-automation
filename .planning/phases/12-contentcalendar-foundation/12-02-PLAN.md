```python
class ContentCalendar:
    def __init__(self):
        self.enabled: bool  # Config.CONTENT_CALENDAR_ENABLED
        self.calendar_path: Path  # Config.TOPIC_DATA_DIR / "content_calendar.json"

    def plan_episode(self, episode_number, release_date, analysis,
                     video_clip_paths=None, full_episode_video_path=None) -> dict:
        """Generate 5-slot calendar for episode. Idempotent — skips if exists."""

    def get_pending_slots(self, episode_data: dict) -> list[dict]:
        """Return slots where status=='pending' and scheduled_at <= now."""

    def mark_slot_uploaded(self, episode_key, slot_name, upload_results) -> None:
        """Set slot status to 'uploaded' with results."""

    def mark_slot_failed(self, episode_key, slot_name, error) -> None:
        """Set slot status to 'failed' with error."""

    def load_all(self) -> dict:
        """Load entire calendar. Returns {} if file missing."""

    def save(self, data: dict) -> None:
        """Atomic write via .tmp + replace."""

    def get_calendar_display(self, episode_number, release_date) -> list[dict]:
        """Return slot dicts with label/dt/type/platforms for dry-run display."""
```
